Online Algorithms

Genereal Framework

  • predict a vector $w_t \in S$

  • receive a convex loss function $f_t: S \rightarrow R$

  • suffer loss $f_t(\mathbf{w}_t)$

Follow the Regularized Leader Mantra

$$\mathbf{w}_{t+1} = arg \min_w \bigg\{g_{1:t} \cdot w + \alpha_{1:t}\psi(w) + r_{0:t}(w)\bigg\}$$

FTRL-proximal¶

Formulation

$$w^{t+1} = \underset{w}{\mathrm{argmin}} \bigg\{g^{(1:t)} \cdot w + \lambda_1\|w\|_1 + \lambda_2\|w\|_2^2 + \frac{1}{2}\sum_{s=1}^t \sigma^{s}\|w-w^{(s)}\|_2^2\bigg\}$$$$\text{where } \sigma^{(s)}_i = \frac{1}{\alpha}\bigg(\sqrt{\sum_{j=1}^s (g_i^{(j)})^2} - \sqrt{\sum_{j=1}^{s-1} (g_i^{(j)})^2}\bigg)$$

Algorithm

$$w_{i}^{t+1} = \begin{cases} 0 \hspace{40mm} \text{ if $\lvert z_{t,i} \rvert \le 0$} \\[2ex] -\eta_t(z_{t,i} - \text{sgn}(z_{t,i}\lambda_1) \hspace{7mm} \text{ otherwise} \end{cases}$$

Implementation

import numpy as np

class FTRLProximal:
    def __init__(self, alpha, beta, L1, L2, n_inputs = 1):
        self.alpha = alpha
        self.alpha_inv = 1 / alpha
        self.beta = beta
        self.L1 = L1
        self.L2 = L2
        self.z = np.zeros(n_inputs)
        self.n = np.zeros(n_inputs)
        
    def predict(self, idx, x):
        w = self.__weight_update(idx)
        return self.__sigmoid(np.dot(w, x)) 
    
    def fit(self, idx, x, y):       
        w = self.__weight_update(idx)        
        p = self.__sigmoid(np.dot(w, x))
        
        g = (p - y) * x
        dn = self.n[idx] + np.power(g, 2)     
        sigma = self.alpha_inv * (np.sqrt(dn) - np.sqrt(self.n[idx]))
        
        self.z[idx] = self.z[idx] + g - np.multiply(sigma, w)
        self.n[idx] = dn       
        return p
    
    def __weight_update(self, idx):    
        dw = np.zeros(idx.size)
        mask = np.abs(self.z[idx]) > self.L1        
        
        z_i = self.z[idx][mask]
        n_i = self.n[idx][mask]        
        tmp_1 = z_i - np.sign(z_i) * self.L1
        tmp_2 = (self.beta + np.sqrt(n_i)) * self.alpha_inv + self.L2       
        
        dw[mask] = -np.divide(tmp_1, tmp_2)         
        return dw
        
    def __sigmoid(self, x):
        return 1 / (1 + np.exp(-x))

Time-decaying Adaptive Prediction¶

Formulation

$$w_{t+1} = \underset{w}{\mathrm{argmin}} \bigg\{g^{(1:t)} \cdot w + \lambda_1\|w\|_1 + \lambda_2\|w\|_2^2 + \frac{1}{2}\sum_{s=1}^t \delta^{s,t}\|w-w^{(s)}\|_2^2\bigg\}$$$$\text{where } \delta^{(s,t)}_i = \sigma^{(s)}_i\exp{(-\frac{\lvert t+1-s\rvert}{(2\tau)^2})}$$

Algorithm

Implementation

import numpy as np

class TDAP:
    def __init__(self, alpha, lamb, L1, L2, n_inputs = 1):
        self.alpha_inv = 1 / alpha
        self.exp_lambda = np.exp(-lamb)
        self.L1 = L1
        self.L2 = L2
        self.u = np.zeros(n_inputs)
        self.v = np.zeros(n_inputs)
        self.h = np.zeros(n_inputs)
        self.z = np.zeros(n_inputs)
        self.delta = np.zeros(n_inputs)
    
    def fit(self, idx, x, y):       
        w = self.__weight_update(idx)        
        p = self.__sigmoid(np.dot(w, x))
        g = (p - y) * x
        new_u = self.u[idx] + np.square(g)
        sigma = self.alpha_inv * (np.sqrt(new_u) - np.sqrt(self.u[idx]))
        
        self.u[idx] = new_u
        self.delta[idx] = self.exp_lambda * (self.delta[idx] + sigma)
        self.v[idx] = self.v[idx] + g
        self.h[idx] = self.exp_lambda * (self.h[idx] + np.multiply(sigma, w))
        self.z[idx] = self.v[idx] - self.h[idx]
        return p
    
    def __weight_update(self, idx):    
        dw = np.zeros(idx.size)
        mask = np.abs(self.z[idx]) > self.L1    
        
        z_i     = self.z[idx][mask]
        delta_i = self.delta[idx][mask]
        
        tmp_1 = (self.L2 + delta_i)
        tmp_2 = z_i - self.L1 * np.sign(z_i)
        
        dw[mask] = -np.divide(tmp_2, tmp_1)         
        return dw
    
    def predict(self, idx, x):
        w = self.__weight_update(idx)
        return self.__sigmoid(np.dot(w, x)) 
        
    def __sigmoid(self, x):
        return 1 / (1 + np.exp(-x))

Online Gradient Descent¶

Derivation

$$w_{t+1} = arg \min_w \bigg\{g_{1:t} \cdot w + \frac{1}{2\eta}\|w\|^2_2\bigg\}$$

Algorithm

$$w_{t+1} = w_t - \eta g_t$$

Implementation

import numpy as np

class OGD:
    def __init__(self, eta, n_inputs = 1):
        self.eta = eta
        self.w = np.zeros(n_inputs)
    
    def fit(self, x, y):       
        p = self.predict(x)
        self.w = self.w - self.eta * (p-y) * x
        return p
    
    def predict(self, x):
        return self.__sigmoid(np.dot(self.w, x)) 
        
    def __sigmoid(self, x):
        return 1 / (1 + np.exp(-x))

Regularized Dual Average¶

Derivation

$$w_{t+1} = arg \min_w \bigg\{\frac{1}{t} g_{1:t} \cdot w + \lambda\|w\|_1 + \frac{\gamma}{\sqrt{t}} \frac{1}{2}\|w\|^2_2\bigg\}$$

Algorithm

$$w = \begin{cases} 0 \text{ if $\lvert\eta_t\rvert \le \lambda$} \\[2ex] -\frac{\sqrt{t}}{\gamma}(\eta_t - \lambda sgn(\eta_t)) \text{ otherwise} \end{cases} \text{ where } \eta_t = \frac{1}{t}g_{1:t}$$

Implementation

import numpy as np

class RDA:
    def __init__(self, L1, L2, n_inputs = 1):
        self.L1 = L1
        self.L2 = L2
        self.counter = 0
        self.eta = np.zeros(n_inputs)
    
    def fit(self, idx, x, y):   
        self.counter += 1
        w = self.weight_update(idx)        
        p = self.__sigmoid(np.dot(w, x))
        
        g = (p - y) * x
        self.eta[idx] = ((self.counter-1)*self.eta[idx] + g)/self.counter
        return p
    
    def weight_update(self, idx):    
        dw = np.zeros(idx.size)
        mask = np.abs(self.eta[idx]) > self.L1    
        dw[mask] = (self.L1*np.sign(self.eta[idx][mask])-self.eta[idx][mask])*np.sqrt(self.counter)/self.L2
        return dw
    
    def predict(self, idx, x):
        w = self.weight_update(idx)
        return self.__sigmoid(np.dot(w, x)) 
        
    def __sigmoid(self, x):
        return 1 / (1 + np.exp(-x))

def append_error(algorithm, error):
    if algorithm in errors:
        errors[algorithm] = np.append(errors[algorithm], error)
    else:
        errors[algorithm] = np.array(error)

Evaluate Algorithms

ogd = OGD(eta=0.5, n_inputs=X_input.shape[1])
ftrl_proximal = FTRLProximal(alpha=0.3, beta=0.0, L1=1.0, L2=1.0, n_inputs=X_input.shape[1])
tdap = TDAP(alpha=0.3, lamb = 0.00001, L1=1.0, L2=1.0, n_inputs=X_input.shape[1])
rda = RDA(L1=1.0, L2=1.0, n_inputs=X_input.shape[1])

errors = {}
indices = np.array(range(X_input.shape[1]))
        
for row in range(len(Y_label)):
    x = X_input[row]
    y = Y_label[row]-1

    # perceptron
    p = ogd.fit(x, y)
    error = [int(np.abs(y-p)>0.5)]
    append_error('ogd', error)
        
    # ftrl_proximal
    p = ftrl_proximal.fit(indices, x, y)
    error = [int(np.abs(y-p)>0.5)]
    append_error('ftrl_proximal', error)
    
    # tdap
    p = tdap.fit(indices, x, y)
    error = [int(np.abs(y-p)>0.5)]
    append_error('tdap', error)
    
    # tdap
    p = rda.fit(indices, x, y)
    error = [int(np.abs(y-p)>0.5)]
